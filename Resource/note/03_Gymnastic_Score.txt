a,b,c,d=[float(i) for i in input().split()]
min,max=0,0
if a <= b and a <=c and a <=d:
    min=a
elif b <= c and b <=a and b <=d:
    min=b
elif c <= a and c <=b and c <=d:
    min=c
else: min=d
if a >= b and a >=c and a >=d:
    max=a
elif b >= c and b >=a and b >=d:
    max=b
elif c >= a and c >=b and c >=d:
    max=c
else: max=d
print(round((a+b+c+d-max-min)/2,2))
