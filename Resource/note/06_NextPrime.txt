def is_prime(n):
    if n<=1:
        return False
    for k in range(2,int(n**0.5)+1):
        if n%k==0:
            return False
    return True

def next_prime(N):
    if is_prime(N+1):
        return N+1
    else: return next_prime(N+1)

def next_twin_prime(N):
    if next_prime(next_prime(N))-next_prime(N)==2:
        return (next_prime(N),next_prime(next_prime(N)))
    else: return next_twin_prime(next_prime(N))
