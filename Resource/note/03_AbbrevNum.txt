n=int(input())
if n>=10000000000:
    print(str(round(n/1000000000))+'B')
elif n>=1000000000:
    print(str(round(n/100000000)/10)+'B')
elif n>=10000000:
    print(str(round(n/1000000))+'M')
elif n>=1000000:
    print(str(round(n/100000)/10)+'M')
elif n>=10000:
    print(str(round(n/1000))+'K')
elif n>=1000:
    print(str(round(n/100)/10)+'K')
else:print(n)
