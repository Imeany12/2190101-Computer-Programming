def rot13(n):
    s=""
    for i in n:
        if 65<=ord(i)<=90:
            s+=chr((ord(i)-65+13)%26+65)
        elif 97<=ord(i)<=122:
            s+=chr((ord(i)-97+13)%26+97)
        else:s+=i
    return s
word=[]
while True:
    a=input()
    if a=="end":
        break
    word+=[rot13(a)]
print("\n".join(word))07_
