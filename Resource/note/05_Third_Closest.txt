import math
n=int(input())
x=[]
y=[]
c=[]
d=[]
for i in range(n):
    a,b=input().split()
    x=x+[float(a)]
    y=y+[float(b)]
    c=c+[math.sqrt(float(a)**2+float(b)**2)]
    d=d+[math.sqrt(float(a)**2+float(b)**2)]
c.sort()
index=d.index(c[2])
print('#'+str(index+1)+':','('+str(x[index])+',',str(y[index])+')')
