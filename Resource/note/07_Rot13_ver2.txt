def rot13(x):
    upper = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
    lower = "abcdefghijklmnopqrstuvwxyz"
    output = ""
    for i in x:
        if i in upper:
            output += upper[(upper.find(i) + 13) % 26]
        elif i in lower:
            output += lower[(lower.find(i) + 13) % 26]
        else:
            output += i
    return output

s = input()
while s != "end": 
    print(rot13(s))
    s = input()
